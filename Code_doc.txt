spark.sql(f"""
CREATE OR REPLACE TEMP VIEW mbna_parameters AS

WITH base_dates AS (
    SELECT
        DATE('{start_date}') AS start_date,
        DATE('{end_date}') AS end_date
)

SELECT
    start_date,
    end_date,

    TO_DATE(
        DATE_TRUNC(
            'MONTH',
            ADD_MONTHS(end_date, -1)
        )
    ) AS previous_month_start,

    ADD_MONTHS(end_date, -1) AS previous_month_end,

    TO_DATE(
        DATE_TRUNC('MONTH', end_date)
    ) AS current_mtd_start,

    DATE_SUB(end_date, 6) AS week_start,

    end_date AS week_end

FROM base_dates
""")


%sql

CREATE OR REPLACE TEMP VIEW mbna_application_base AS

WITH filtered_applications AS (
    SELECT
        CAST(a.application_number AS STRING) AS application_number,
        CAST(a.account_number AS STRING) AS account_number,

        TO_DATE(a.load_date) AS load_date,
        TO_DATE(a.application_date) AS application_date,
        TO_DATE(a.date_account_opened) AS date_account_opened,

        CAST(a.transit_number AS STRING) AS transit_number,
        a.data_source,

        CASE
            WHEN a.date_account_opened IS NULL
                OR a.application_date IS NULL
            THEN 'UNKNOWN'

            WHEN DATEDIFF(
                TO_DATE(a.application_date),
                TO_DATE(a.date_account_opened)
            ) <= 90
            THEN 'N2B'

            ELSE 'E2B'
        END AS customer_tenure_90d,

        ROW_NUMBER() OVER (
            PARTITION BY CAST(a.application_number AS STRING)
            ORDER BY
                a.load_date DESC,
                a.application_date DESC
        ) AS row_num

    FROM hive_metastore.fiqm.applications a
    CROSS JOIN mbna_parameters p

    WHERE
        (
            UPPER(TRIM(a.data_source)) = 'MBNA'
            OR CAST(a.application_number AS STRING) RLIKE '^[A-F]'
        )

        AND a.load_date >= p.start_date
        AND a.load_date < DATE_ADD(p.end_date, 1)
)

SELECT
    application_number,
    account_number,
    load_date,
    application_date,
    date_account_opened,
    transit_number,
    data_source,
    customer_tenure_90d

FROM filtered_applications

WHERE row_num = 1;

%sql

CREATE OR REPLACE TEMP VIEW mbna_tsys_base AS

SELECT
    CAST(t.applid AS STRING) AS applid,
    CAST(t.acct_num_ta1 AS STRING) AS card_number,
    TO_DATE(t.open_dt) AS card_open_date,

    t.stc_secure_fraud,
    TO_DATE(t.sec_fraud_stat_dt) AS secure_fraud_status_date

FROM tsys.v_cfm_tsys_mbna_acct_slim t;


%sql

CREATE OR REPLACE TEMP VIEW mbna_application_cards AS

WITH card_candidates AS (
    SELECT
        app.application_number,
        app.account_number,
        app.load_date,
        app.application_date,
        app.date_account_opened,
        app.transit_number,
        app.customer_tenure_90d,

        tsys.applid,
        tsys.card_number,
        tsys.card_open_date,

        ROW_NUMBER() OVER (
            PARTITION BY
                app.application_number,
                tsys.card_number

            ORDER BY
                ABS(
                    DATEDIFF(
                        tsys.card_open_date,
                        app.application_date
                    )
                ),
                tsys.card_open_date
        ) AS row_num

    FROM mbna_application_base app

    LEFT JOIN mbna_tsys_base tsys
        ON tsys.applid = app.account_number

       AND tsys.card_open_date >= DATE_SUB(
            app.application_date,
            7
       )

       AND tsys.card_open_date <= DATE_ADD(
            app.application_date,
            30
       )
)

SELECT
    application_number,
    account_number,
    load_date,
    application_date,
    date_account_opened,
    transit_number,
    customer_tenure_90d,
    applid,
    card_number,
    card_open_date

FROM card_candidates

WHERE row_num = 1;

%sql

CREATE OR REPLACE TEMP VIEW mbna_application_actions AS

WITH filtered_actions AS (
    SELECT
        CAST(act.application_number AS STRING) AS application_number,
        TO_DATE(act.date_updated) AS fraud_date,

        ROW_NUMBER() OVER (
            PARTITION BY CAST(act.application_number AS STRING)
            ORDER BY act.date_updated DESC
        ) AS row_num

    FROM hive_metastore.fiqm.tdct_rt_action act
    CROSS JOIN mbna_parameters p

    WHERE
        CAST(act.application_number AS STRING) RLIKE '^[A-F]'

        AND UPPER(TRIM(act.action_status_code)) IN ('H', 'S')

        AND act.date_updated >= p.start_date
        AND act.date_updated < DATE_ADD(p.end_date, 1)
)

SELECT
    application_number,
    fraud_date

FROM filtered_actions

WHERE row_num = 1;


%sql

CREATE OR REPLACE TEMP VIEW mbna_application_fraud_cards AS

SELECT
    cards.application_number,
    cards.transit_number,
    cards.customer_tenure_90d,
    cards.card_number,
    actions.fraud_date,
    'APPLICATION_ACTION' AS fraud_source

FROM mbna_application_cards cards

INNER JOIN mbna_application_actions actions
    ON cards.application_number = actions.application_number

WHERE cards.card_number IS NOT NULL;


%sql

CREATE OR REPLACE TEMP VIEW mbna_tsys_fraud_cards AS

WITH filtered_card_frauds AS (
    SELECT
        applid,
        card_number,

        COALESCE(
            secure_fraud_status_date,
            card_open_date
        ) AS fraud_date,

        ROW_NUMBER() OVER (
            PARTITION BY card_number
            ORDER BY
                COALESCE(
                    secure_fraud_status_date,
                    card_open_date
                ) ASC
        ) AS row_num

    FROM mbna_tsys_base
    CROSS JOIN mbna_parameters p

    WHERE
        stc_secure_fraud IS NOT NULL

        AND COALESCE(
            secure_fraud_status_date,
            card_open_date
        ) >= p.start_date

        AND COALESCE(
            secure_fraud_status_date,
            card_open_date
        ) < DATE_ADD(p.end_date, 1)
)

SELECT
    applid,
    card_number,
    fraud_date

FROM filtered_card_frauds

WHERE row_num = 1;


%sql

CREATE OR REPLACE TEMP VIEW mbna_direct_card_frauds AS

SELECT
    app.application_number,
    app.transit_number,
    app.customer_tenure_90d,
    fraud.card_number,
    fraud.fraud_date,
    'TSYS_CARD_FRAUD' AS fraud_source

FROM mbna_tsys_fraud_cards fraud

LEFT JOIN mbna_application_base app
    ON fraud.applid = app.account_number

WHERE app.application_number IS NOT NULL;



%sql

CREATE OR REPLACE TEMP VIEW mbna_fraud_cards_base AS

WITH combined_frauds AS (
    SELECT
        application_number,
        transit_number,
        customer_tenure_90d,
        card_number,
        fraud_date,
        fraud_source

    FROM mbna_application_fraud_cards

    UNION ALL

    SELECT
        application_number,
        transit_number,
        customer_tenure_90d,
        card_number,
        fraud_date,
        fraud_source

    FROM mbna_direct_card_frauds
),

canonical_fraud AS (
    SELECT
        application_number,
        transit_number,
        customer_tenure_90d,
        card_number,

        MIN(fraud_date) AS fraud_date,

        CONCAT_WS(
            ',',
            SORT_ARRAY(
                COLLECT_SET(fraud_source)
            )
        ) AS fraud_sources

    FROM combined_frauds

    GROUP BY
        application_number,
        transit_number,
        customer_tenure_90d,
        card_number
)

SELECT *
FROM canonical_fraud;

CREATE OR REPLACE TEMP VIEW mbna_fraud_cards AS

SELECT

    fraud.transit_number,
    fraud.customer_tenure_90d,

WITH application_periods AS (

    -- ========================================================
    -- MONTHLY
    -- ========================================================
    SELECT
        'MONTHLY' AS reporting_period,
        DATE_FORMAT(load_date, 'yyyy-MM') AS period_label,

        COUNT(DISTINCT application_number) AS num_apps,
        COUNT(DISTINCT card_number) AS num_cards

    FROM mbna_application_cards

    GROUP BY DATE_FORMAT(load_date, 'yyyy-MM')


    UNION ALL


    -- ========================================================
    -- PREVIOUS MONTH MTD
    -- ========================================================
    SELECT
        'PREVIOUS_MONTH_MTD' AS reporting_period,

        CONCAT(
            DATE_FORMAT(p.previous_month_start, 'yyyy-MM-dd'),
            ' to ',
            DATE_FORMAT(p.previous_month_end, 'yyyy-MM-dd')
        ) AS period_label,

        COUNT(DISTINCT d.application_number) AS num_apps,
        COUNT(DISTINCT d.card_number) AS num_cards

    FROM mbna_application_cards d
    CROSS JOIN mbna_parameters p

    WHERE
        d.load_date >= p.previous_month_start
        AND d.load_date < DATE_ADD(p.previous_month_end, 1)

    GROUP BY
        p.previous_month_start,
        p.previous_month_end


    UNION ALL


    -- ========================================================
    -- CURRENT MTD
    -- ========================================================
    SELECT
        'CURRENT_MTD' AS reporting_period,

        CONCAT(
            DATE_FORMAT(p.current_mtd_start, 'yyyy-MM-dd'),
            ' to ',
            DATE_FORMAT(p.end_date, 'yyyy-MM-dd')
        ) AS period_label,

        COUNT(DISTINCT d.application_number) AS num_apps,
        COUNT(DISTINCT d.card_number) AS num_cards

    FROM mbna_application_cards d
    CROSS JOIN mbna_parameters p

    WHERE
        d.load_date >= p.current_mtd_start
        AND d.load_date < DATE_ADD(p.end_date, 1)

    GROUP BY
        p.current_mtd_start,
        p.end_date


    UNION ALL


    -- ========================================================
    -- WEEKLY
    -- ========================================================
    SELECT
        'WEEKLY' AS reporting_period,

        CONCAT(
            DATE_FORMAT(p.week_start, 'yyyy-MM-dd'),
            ' to ',
            DATE_FORMAT(p.week_end, 'yyyy-MM-dd')
        ) AS period_label,

        COUNT(DISTINCT d.application_number) AS num_apps,
        COUNT(DISTINCT d.card_number) AS num_cards

    FROM mbna_application_cards d
    CROSS JOIN mbna_parameters p

    WHERE
        d.load_date >= p.week_start
        AND d.load_date < DATE_ADD(p.week_end, 1)

    GROUP BY
        p.week_start,
        p.week_end
),

fraud_periods AS (

    UNION ALL


    -- ========================================================
    -- CURRENT MTD
    -- ========================================================
    SELECT
        'CURRENT_MTD' AS reporting_period,

        CONCAT(
            DATE_FORMAT(p.current_mtd_start, 'yyyy-MM-dd'),

SELECT
    apps.reporting_period,
    apps.period_label,

    apps.num_apps,
    apps.num_cards,

    COALESCE(fraud.num_fraud_apps, 0) AS num_fraud_apps,
    COALESCE(fraud.num_fraud_cards, 0) AS num_fraud_cards,

    COALESCE(
        fraud.num_alerted_fraud_apps,
        0
    ) AS num_alerted_fraud_apps,

    COALESCE(
        fraud.num_alerted_fraud_cards,
        0
    ) AS num_alerted_fraud_cards,

    ROUND(
        100.0 * COALESCE(fraud.num_fraud_apps, 0)
        / NULLIF(apps.num_apps, 0),
        2
    ) AS application_fraud_rate_pct,

    ROUND(
        100.0 * COALESCE(fraud.num_fraud_cards, 0)
        / NULLIF(apps.num_cards, 0),
        2
    ) AS card_fraud_rate_pct,

    ROUND(
        100.0 * COALESCE(fraud.num_alerted_fraud_cards, 0)
        / NULLIF(fraud.num_fraud_cards, 0),
        2
    ) AS alerted_fraud_capture_rate_pct

FROM application_periods apps

LEFT JOIN fraud_periods fraud
    ON apps.reporting_period = fraud.reporting_period
   AND apps.period_label = fraud.period_label

ORDER BY
    CASE apps.reporting_period
        WHEN 'MONTHLY' THEN 1
        WHEN 'PREVIOUS_MONTH_MTD' THEN 2
        WHEN 'CURRENT_MTD' THEN 3
        WHEN 'WEEKLY' THEN 4
        ELSE 5
    END,
    apps.period_label;

        / NULLIF(apps.num_apps, 0),
        2
    ) AS application_fraud_rate_pct,

    ROUND(
        100.0 * COALESCE(fraud.num_fraud_cards, 0)
        / NULLIF(apps.num_cards, 0),
        2
    ) AS card_fraud_rate_pct,

    ROUND(
        100.0 * COALESCE(fraud.num_alerted_fraud_cards, 0)
        / NULLIF(fraud.num_fraud_cards, 0),
        2
    ) AS alerted_fraud_capture_rate_pct

FROM application_periods apps

LEFT JOIN fraud_periods fraud
    ON apps.reporting_period = fraud.reporting_period
   AND apps.period_label = fraud.period_label
   AND apps.customer_tenure_90d = fraud.customer_tenure_90d

ORDER BY
    CASE apps.reporting_period
        WHEN 'MONTHLY' THEN 1
        WHEN 'PREVIOUS_MONTH_MTD' THEN 2
        WHEN 'CURRENT_MTD' THEN 3
        WHEN 'WEEKLY' THEN 4
        ELSE 5
    END,

    apps.period_label,

    CASE apps.customer_tenure_90d
        WHEN 'N2B' THEN 1
        WHEN 'E2B' THEN 2
        WHEN 'UNKNOWN' THEN 3
        ELSE 4
    END;

WITH application_periods AS (

    SELECT
        'MONTHLY' AS reporting_period,
        DATE_FORMAT(load_date, 'yyyy-MM') AS period_label,
        COALESCE(transit_number, 'UNKNOWN') AS transit_number,

        COUNT(DISTINCT application_number) AS num_apps,
        COUNT(DISTINCT card_number) AS num_cards

    FROM mbna_application_cards

    GROUP BY
        DATE_FORMAT(load_date, 'yyyy-MM'),
        COALESCE(transit_number, 'UNKNOWN')


    UNION ALL


    SELECT
        'PREVIOUS_MONTH_MTD' AS reporting_period,

        CONCAT(
            DATE_FORMAT(p.previous_month_start, 'yyyy-MM-dd'),
            ' to ',
            DATE_FORMAT(p.previous_month_end, 'yyyy-MM-dd')
        ) AS period_label,

        COALESCE(d.transit_number, 'UNKNOWN') AS transit_number,

        COUNT(DISTINCT d.application_number) AS num_apps,
        COUNT(DISTINCT d.card_number) AS num_cards

    FROM mbna_application_cards d
    CROSS JOIN mbna_parameters p

    WHERE
        d.load_date >= p.previous_month_start
        AND d.load_date < DATE_ADD(p.previous_month_end, 1)

    GROUP BY
        p.previous_month_start,
        p.previous_month_end,
        COALESCE(d.transit_number, 'UNKNOWN')


    UNION ALL


    SELECT
        'CURRENT_MTD' AS reporting_period,

        CONCAT(
            DATE_FORMAT(p.current_mtd_start, 'yyyy-MM-dd'),
            ' to ',
            DATE_FORMAT(p.end_date, 'yyyy-MM-dd')
        ) AS period_label,

        COALESCE(d.transit_number, 'UNKNOWN') AS transit_number,

        COUNT(DISTINCT d.application_number) AS num_apps,
        COUNT(DISTINCT d.card_number) AS num_cards

    FROM mbna_application_cards d
    CROSS JOIN mbna_parameters p

    WHERE
        d.load_date >= p.current_mtd_start
        AND d.load_date < DATE_ADD(p.end_date, 1)

    GROUP BY
        p.current_mtd_start,
        p.end_date,
        COALESCE(d.transit_number, 'UNKNOWN')


    UNION ALL


    SELECT
        'WEEKLY' AS reporting_period,

        CONCAT(
            DATE_FORMAT(p.week_start, 'yyyy-MM-dd'),
            ' to ',
            DATE_FORMAT(p.week_end, 'yyyy-MM-dd')
        ) AS period_label,

        COALESCE(d.transit_number, 'UNKNOWN') AS transit_number,

        COUNT(DISTINCT d.application_number) AS num_apps,
        COUNT(DISTINCT d.card_number) AS num_cards

    FROM mbna_application_cards d
    CROSS JOIN mbna_parameters p

    WHERE
        d.load_date >= p.week_start
        AND d.load_date < DATE_ADD(p.week_end, 1)

    GROUP BY
        p.week_start,
        p.week_end,
        COALESCE(d.transit_number, 'UNKNOWN')
),
