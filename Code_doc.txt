import pandas as pd

mapping_path = "/Workspace/Users/yash.sethi@td.com/IFAD to DSAP Script Migration/TDAF_queue_rules.csv"

pdf = pd.read_csv(mapping_path)

pdf = (
    pdf[["Queue", "rule_name"]]
    .dropna(subset=["Queue", "rule_name"])
    .drop_duplicates()
)

df_rules = spark.createDataFrame(pdf)
df_rules.createOrReplaceTempView("rc_tdaf_rules_raw")

CREATE OR REPLACE TEMP VIEW params AS
SELECT
    add_months(date_trunc('MONTH', current_date()), -1) AS start_date,
    date_trunc('MONTH', current_date()) AS end_date
;

CREATE OR REPLACE TEMP VIEW rc_tdaf_rules AS
SELECT
    lpad(CAST(dense_rank() OVER (ORDER BY Queue) AS STRING), 2, '0') AS queue_priority,
    Queue AS queue_name,
    rule_name
FROM rc_tdaf_rules_raw
WHERE Queue IS NOT NULL
  AND rule_name IS NOT NULL
;

CREATE OR REPLACE TEMP VIEW rc_actions_before AS
SELECT
    c.application_number,
    MAX(c.date_updated) AS date_updated
FROM hive_metastore.fiqm.tdct_rt_action c
CROSS JOIN params p
WHERE c.date_updated < p.start_date
GROUP BY c.application_number
;

CREATE OR REPLACE TEMP VIEW rc_actions AS
SELECT
    c.application_number,
    MAX(c.date_updated) AS date_updated
FROM hive_metastore.fiqm.tdct_rt_action c
CROSS JOIN params p
WHERE c.date_updated >= p.start_date
  AND NOT EXISTS (
      SELECT 1
      FROM rc_actions_before d
      WHERE d.application_number = c.application_number
  )
GROUP BY c.application_number
;

CREATE OR REPLACE TEMP VIEW rc_tdaf_apps AS
SELECT
    trim(application_number) AS new_application_number,
    *
FROM hive_metastore.fiqm.applications
WHERE product_type = 'AL'
  AND data_source = 'Indirect'
;

CREATE OR REPLACE TEMP VIEW rc_temp_tdaf_rulehits AS
SELECT
    a.*,
    CAST(a.create_date_time AS DATE) AS alert_date
FROM hive_metastore.fiqm.tdct_rt_match a
INNER JOIN rc_tdaf_apps b
    ON a.application_number = b.new_application_number
CROSS JOIN params p
WHERE a.rule_name NOT LIKE 'W_%'
  AND a.create_date_time >= p.start_date
  AND a.create_date_time < p.end_date
;

CREATE OR REPLACE TEMP VIEW rc_temp_tdaf_rulehits_2 AS
SELECT
    a.*,
    b.queue_priority,
    b.queue_name
FROM rc_temp_tdaf_rulehits a
LEFT JOIN rc_tdaf_rules b
    ON a.rule_name = b.rule_name
WHERE b.rule_name IS NOT NULL
;

CREATE OR REPLACE TEMP VIEW rc_temp_tdaf_rulehits_3 AS
SELECT
    application_number,
    CAST(create_date_time AS DATE) AS create_date_time,
    MIN(queue_priority) AS final_queue_priority
FROM rc_temp_tdaf_rulehits_2
GROUP BY application_number, CAST(create_date_time AS DATE)
;

CREATE OR REPLACE TEMP VIEW rc_temp_tdaf_rulehits_4 AS
SELECT DISTINCT
    a.application_number,
    a.create_date_time,
    a.final_queue_priority,
    b.queue_name AS temp_task_name
FROM rc_temp_tdaf_rulehits_3 a
LEFT JOIN rc_tdaf_rules b
    ON a.final_queue_priority = b.queue_priority
;

CREATE OR REPLACE TEMP VIEW rc_tdaf_rulehits_base AS
SELECT
    a.*,
    b.temp_task_name,
    CASE
        WHEN a.task_name IS NOT NULL AND a.task_name <> '' THEN a.task_name
        ELSE b.temp_task_name
    END AS new_task_name
FROM rc_temp_tdaf_rulehits a
LEFT JOIN rc_temp_tdaf_rulehits_4 b
    ON a.application_number = b.application_number
   AND CAST(a.create_date_time AS DATE) = b.create_date_time
WHERE b.temp_task_name IS NOT NULL
;

CREATE OR REPLACE TEMP VIEW rc_queue_performance_temp AS
SELECT
    b.new_task_name,

    COUNT(DISTINCT CONCAT(CAST(a.new_application_number AS STRING), '_', date_format(b.create_date_time, 'yyyy-MM-dd'))) AS queue_alert,

    COUNT(DISTINCT CASE
        WHEN d.action_status_code IN ('CA','CN','H','S','P')
        THEN CONCAT(CAST(a.new_application_number AS STRING), '_', date_format(b.create_date_time, 'yyyy-MM-dd'))
    END) AS queue_worked,

    COUNT(DISTINCT CASE WHEN d.action_status_code IN ('CN')
        THEN CONCAT(CAST(a.new_application_number AS STRING), '_', date_format(b.create_date_time, 'yyyy-MM-dd')) END) AS queue_clear_noinvestigation,

    COUNT(DISTINCT CASE WHEN d.action_status_code IN ('P')
        THEN CONCAT(CAST(a.new_application_number AS STRING), '_', date_format(b.create_date_time, 'yyyy-MM-dd')) END) AS queue_possible_fraud,

    COUNT(DISTINCT CASE WHEN d.action_status_code IN ('CA')
        THEN CONCAT(CAST(a.new_application_number AS STRING), '_', date_format(b.create_date_time, 'yyyy-MM-dd')) END) AS queue_clear_after_investigation,

    COUNT(DISTINCT CASE WHEN d.action_status_code IN ('I')
        THEN CONCAT(CAST(a.new_application_number AS STRING), '_', date_format(b.create_date_time, 'yyyy-MM-dd')) END) AS queue_under_investigation,

    COUNT(DISTINCT CASE WHEN d.action_status_code IN ('P','CA','H','S')
        THEN CONCAT(CAST(a.new_application_number AS STRING), '_', date_format(b.create_date_time, 'yyyy-MM-dd')) END) AS queue_fully_investigated,

    COUNT(DISTINCT CASE WHEN d.action_status_code IN ('H','S')
        THEN CONCAT(CAST(a.new_application_number AS STRING), '_', date_format(b.create_date_time, 'yyyy-MM-dd')) END) AS queue_confirmed_fraud,

    COUNT(DISTINCT a.new_application_number) AS unique_alerted_application,

    COUNT(DISTINCT CASE WHEN d.action_status_code IN ('CA','CN','H','S','P') THEN a.new_application_number END) AS customer_worked,
    COUNT(DISTINCT CASE WHEN d.action_status_code IN ('CN') THEN a.new_application_number END) AS customer_clear_noinvestigation,
    COUNT(DISTINCT CASE WHEN d.action_status_code IN ('P') THEN a.new_application_number END) AS customer_possible_fraud,
    COUNT(DISTINCT CASE WHEN d.action_status_code IN ('CA') THEN a.new_application_number END) AS customer_clear_after_investigation,
    COUNT(DISTINCT CASE WHEN d.action_status_code IN ('I') THEN a.new_application_number END) AS customer_clear_under_investigation,
    COUNT(DISTINCT CASE WHEN d.action_status_code IN ('P','CA','H','S') THEN a.new_application_number END) AS customer_fully_investigated,
    COUNT(DISTINCT CASE WHEN d.action_status_code IN ('H','S') THEN a.new_application_number END) AS customer_confirmed_fraud

FROM rc_tdaf_apps a
LEFT JOIN rc_tdaf_rulehits_base b
    ON a.new_application_number = b.application_number
LEFT JOIN rc_actions c
    ON a.new_application_number = c.application_number
LEFT JOIN hive_metastore.fiqm.tdct_rt_action d
    ON c.application_number = d.application_number
   AND c.date_updated = d.date_updated
LEFT JOIN rc_tdaf_rules e
    ON b.rule_name = e.rule_name
CROSS JOIN params p
WHERE a.product_type = 'AL'
  AND a.data_source = 'Indirect'
  AND e.rule_name IS NOT NULL
  AND b.create_date_time >= p.start_date
  AND b.create_date_time < p.end_date
GROUP BY b.new_task_name
;

CREATE OR REPLACE TEMP VIEW rc_queue_exposure_temp AS
SELECT
    t.new_task_name,
    COUNT(DISTINCT t.new_application_number) AS fraud_count,
    SUM(t.saved_value) AS averted_exposure
FROM (
    SELECT DISTINCT
        b.new_task_name,
        a.new_application_number,
        d.saved_value
    FROM rc_tdaf_apps a
    LEFT JOIN rc_tdaf_rulehits_base b
        ON a.new_application_number = b.application_number
    LEFT JOIN rc_actions c
        ON a.new_application_number = c.application_number
    LEFT JOIN hive_metastore.fiqm.tdct_rt_action d
        ON c.application_number = d.application_number
       AND c.date_updated = d.date_updated
    LEFT JOIN rc_tdaf_rules e
        ON b.rule_name = e.rule_name
    CROSS JOIN params p
    WHERE a.product_type = 'AL'
      AND a.data_source = 'Indirect'
      AND e.rule_name IS NOT NULL
      AND b.create_date_time >= p.start_date
      AND b.create_date_time < p.end_date
      AND d.action_status_code IN ('H','S')
) t
GROUP BY t.new_task_name
;

CREATE OR REPLACE TEMP VIEW rc_tier_performance_temp AS
SELECT
    CASE
        WHEN b.new_task_name LIKE '%T1%' THEN 'TIER 1'
        WHEN b.new_task_name LIKE '%T2%' THEN 'TIER 2'
        WHEN b.new_task_name LIKE '%T3%' THEN 'TIER 3'
        ELSE 'NOT IN ANY TIER'
    END AS tier,
    SUM(queue_alert) AS queue_alert,
    SUM(queue_worked) AS queue_worked,
    SUM(queue_clear_noinvestigation) AS queue_clear_noinvestigation,
    SUM(queue_possible_fraud) AS queue_possible_fraud,
    SUM(queue_clear_after_investigation) AS queue_clear_after_investigation,
    SUM(queue_under_investigation) AS queue_under_investigation,
    SUM(queue_fully_investigated) AS queue_fully_investigated,
    SUM(queue_confirmed_fraud) AS queue_confirmed_fraud,
    SUM(unique_alerted_application) AS unique_alerted_application,
    SUM(customer_worked) AS customer_worked,
    SUM(customer_clear_noinvestigation) AS customer_clear_noinvestigation,
    SUM(customer_possible_fraud) AS customer_possible_fraud,
    SUM(customer_clear_after_investigation) AS customer_clear_after_investigation,
    SUM(customer_clear_under_investigation) AS customer_clear_under_investigation,
    SUM(customer_fully_investigated) AS customer_fully_investigated,
    SUM(customer_confirmed_fraud) AS customer_confirmed_fraud
FROM rc_queue_performance_temp b
GROUP BY
    CASE
        WHEN b.new_task_name LIKE '%T1%' THEN 'TIER 1'
        WHEN b.new_task_name LIKE '%T2%' THEN 'TIER 2'
        WHEN b.new_task_name LIKE '%T3%' THEN 'TIER 3'
        ELSE 'NOT IN ANY TIER'
    END
;

CREATE OR REPLACE TEMP VIEW rc_tier_exposure_temp AS
SELECT
    CASE
        WHEN new_task_name LIKE '%T1%' THEN 'TIER 1'
        WHEN new_task_name LIKE '%T2%' THEN 'TIER 2'
        WHEN new_task_name LIKE '%T3%' THEN 'TIER 3'
        ELSE 'NOT IN ANY TIER'
    END AS tier,
    SUM(fraud_count) AS fraud_count,
    SUM(averted_exposure) AS averted_exposure
FROM rc_queue_exposure_temp
GROUP BY
    CASE
        WHEN new_task_name LIKE '%T1%' THEN 'TIER 1'
        WHEN new_task_name LIKE '%T2%' THEN 'TIER 2'
        WHEN new_task_name LIKE '%T3%' THEN 'TIER 3'
        ELSE 'NOT IN ANY TIER'
    END
;

CREATE OR REPLACE TEMP VIEW rc_total_performance_temp AS
SELECT
    'TOTAL PERFORMANCE' AS new_task_name,
    SUM(queue_alert) AS queue_alert,
    SUM(queue_worked) AS queue_worked,
    SUM(queue_clear_noinvestigation) AS queue_clear_noinvestigation,
    SUM(queue_possible_fraud) AS queue_possible_fraud,
    SUM(queue_clear_after_investigation) AS queue_clear_after_investigation,
    SUM(queue_under_investigation) AS queue_under_investigation,
    SUM(queue_fully_investigated) AS queue_fully_investigated,
    SUM(queue_confirmed_fraud) AS queue_confirmed_fraud,
    SUM(unique_alerted_application) AS unique_alerted_application,
    SUM(customer_worked) AS customer_worked,
    SUM(customer_clear_noinvestigation) AS customer_clear_noinvestigation,
    SUM(customer_possible_fraud) AS customer_possible_fraud,
    SUM(customer_clear_after_investigation) AS customer_clear_after_investigation,
    SUM(customer_clear_under_investigation) AS customer_clear_under_investigation,
    SUM(customer_fully_investigated) AS customer_fully_investigated,
    SUM(customer_confirmed_fraud) AS customer_confirmed_fraud
FROM rc_queue_performance_temp
;

CREATE OR REPLACE TEMP VIEW rc_total_exposure_temp AS
SELECT
    'TOTAL PERFORMANCE' AS new_task_name,
    SUM(fraud_count) AS fraud_count,
    SUM(averted_exposure) AS averted_exposure
FROM rc_queue_exposure_temp
;


-- FINAL QUEUE PERFORMANCE
SELECT
    r.queue_priority,
    a.new_task_name AS queue_name,
    COALESCE(a.queue_alert, 0) AS queue_alert,
    CAST(COALESCE(a.queue_alert, 0) / 20.0 AS DECIMAL(18,1)) AS alerts_per_day,
    COALESCE(a.queue_worked, 0) AS queue_worked,
    COALESCE(a.queue_confirmed_fraud, 0) AS queue_confirmed_fraud,
    COALESCE(a.queue_fully_investigated, 0) AS queue_fully_investigated,
    COALESCE(a.queue_possible_fraud, 0) AS queue_possible_fraud,
    COALESCE(a.queue_clear_after_investigation, 0) AS queue_clear_after_investigation,
    COALESCE(a.queue_clear_noinvestigation, 0) AS queue_clear_noinvestigation,
    COALESCE(a.queue_under_investigation, 0) AS queue_under_investigation,
    COALESCE(a.unique_alerted_application, 0) AS unique_alerted_application,
    COALESCE(a.customer_confirmed_fraud, 0) AS customer_confirmed_fraud,
    COALESCE(a.customer_worked, 0) AS customer_worked,
    COALESCE(a.customer_fully_investigated, 0) AS customer_fully_investigated,
    COALESCE(b.averted_exposure, 0) AS averted_exposure,
    CASE WHEN COALESCE(a.queue_confirmed_fraud, 0) = 0 THEN 'NA'
         ELSE CONCAT(CAST((CAST(a.queue_alert AS DECIMAL(18,2)) / CAST(a.queue_confirmed_fraud AS DECIMAL(18,2))) - 1 AS STRING), ':1')
    END AS queue_fpr,
    CASE WHEN COALESCE(a.queue_confirmed_fraud, 0) = 0 THEN 'NA'
         ELSE CONCAT(CAST((CAST(a.queue_worked AS DECIMAL(18,2)) / CAST(a.queue_confirmed_fraud AS DECIMAL(18,2))) - 1 AS STRING), ':1')
    END AS worked_fpr,
    CASE WHEN COALESCE(a.customer_confirmed_fraud, 0) = 0 THEN 'NA'
         ELSE CONCAT(CAST((CAST(a.customer_fully_investigated AS DECIMAL(18,2)) / CAST(a.customer_confirmed_fraud AS DECIMAL(18,2))) - 1 AS STRING), ':1')
    END AS customer_fpr
FROM rc_queue_performance_temp a
LEFT JOIN rc_queue_exposure_temp b
    ON a.new_task_name = b.new_task_name
LEFT JOIN (
    SELECT queue_name, MIN(queue_priority) AS queue_priority
    FROM rc_tdaf_rules
    GROUP BY queue_name
) r
    ON a.new_task_name = r.queue_name
ORDER BY r.queue_priority, a.new_task_name
;

-- FINAL TIER PERFORMANCE
SELECT
    a.tier,
    COALESCE(a.queue_alert, 0) AS queue_alert,
    CAST(COALESCE(a.queue_alert, 0) / 20.0 AS DECIMAL(18,1)) AS alerts_per_day,
    COALESCE(a.queue_worked, 0) AS queue_worked,
    COALESCE(a.queue_confirmed_fraud, 0) AS queue_confirmed_fraud,
    COALESCE(a.queue_fully_investigated, 0) AS queue_fully_investigated,
    COALESCE(a.unique_alerted_application, 0) AS unique_alerted_application,
    COALESCE(a.customer_confirmed_fraud, 0) AS customer_confirmed_fraud,
    COALESCE(a.customer_worked, 0) AS customer_worked,
    COALESCE(a.customer_fully_investigated, 0) AS customer_fully_investigated,
    COALESCE(b.averted_exposure, 0) AS averted_exposure
FROM rc_tier_performance_temp a
LEFT JOIN rc_tier_exposure_temp b
    ON a.tier = b.tier
ORDER BY a.tier
;

-- FINAL TOTAL PERFORMANCE
SELECT
    a.new_task_name,
    COALESCE(a.queue_alert, 0) AS queue_alert,
    CAST(COALESCE(a.queue_alert, 0) / 20.0 AS DECIMAL(18,1)) AS alerts_per_day,
    COALESCE(a.queue_worked, 0) AS queue_worked,
    COALESCE(a.queue_confirmed_fraud, 0) AS queue_confirmed_fraud,
    COALESCE(a.queue_fully_investigated, 0) AS queue_fully_investigated,
    COALESCE(a.unique_alerted_application, 0) AS unique_alerted_application,
    COALESCE(a.customer_confirmed_fraud, 0) AS customer_confirmed_fraud,
    COALESCE(a.customer_worked, 0) AS customer_worked,
    COALESCE(a.customer_fully_investigated, 0) AS customer_fully_investigated,
    COALESCE(b.averted_exposure, 0) AS averted_exposure
FROM rc_total_performance_temp a
LEFT JOIN rc_total_exposure_temp b
    ON a.new_task_name = b.new_task_name
;
