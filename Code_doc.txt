import pandas as pd
import math
import os

input_file = r"C:\Your\Folder\large_file.xlsx"
output_folder = r"C:\Your\Folder\Split_Files"

os.makedirs(output_folder, exist_ok=True)

# Read Excel
df = pd.read_excel(input_file, dtype=str)

chunk_size = 9999   # less than 10k records

num_files = math.ceil(len(df) / chunk_size)

for i in range(num_files):
    start = i * chunk_size
    end = start + chunk_size

    chunk_df = df.iloc[start:end]

    output_file = os.path.join(
        output_folder,
        f"large_file_part_{i+1}.xlsx"
    )

    chunk_df.to_excel(output_file, index=False)

    print(f"Saved {output_file} with {len(chunk_df)} records")
