import os
import glob

folder_path = r"C:\Your\Folder"

for old_path in glob.glob(os.path.join(folder_path, "*.fulfill")):
    new_path = old_path[:-8] + ".txt"   # Remove '.fulfill' and add '.txt'

    os.rename(old_path, new_path)

    print(f"Renamed: {os.path.basename(old_path)} -> {os.path.basename(new_path)}")
