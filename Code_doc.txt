import win32com.client as win32

OL_FOLDER_INBOX = 6

def get_folder_path(folder_path):
    outlook = win32.gencache.EnsureDispatch("Outlook.Application")
    ns = outlook.Session
    root = ns.GetDefaultFolder(OL_FOLDER_INBOX).Parent
    current = root
    for seg in folder_path.split("/"):
        seg = seg.strip()
        if not seg:
            continue
        current = current.Folders(seg)
    return current

# Test
folder = get_folder_path("Inbox/MBNA test")
print("Folder found:", folder.Name)
print("Number of items:", folder.Items.Count)
