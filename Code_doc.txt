# Lookup dataframe
fraud_lookup = final_w_alert_frauds_df[
    ["application_number", "idxsc100"]
].drop_duplicates()

results = []

for file_name, df in tdct_dfs.items():

    # Merge
    merged = df.merge(
        fraud_lookup,
        on="application_number",
        how="left"
    )

    # Convert score to numeric
    merged["idxsc100"] = pd.to_numeric(
        merged["idxsc100"],
        errors="coerce"
    )

    # Store minimum score
    results.append({
        "file_name": file_name,
        "min_idxsc100": merged["idxsc100"].min()
    })

# Create summary dataframe
min_scores_df = pd.DataFrame(results)

print(min_scores_df)
