import pandas as pd
from datetime import datetime

historical_file = r"C:\Your\Folder\historical_file.csv"

hist_df = pd.read_csv(historical_file, dtype=str)

# Clean blanks
hist_df["card_number"] = (
    hist_df["card_number"]
    .astype("string")
    .str.strip()
    .replace(["", "nan", "None", "NaN"], pd.NA)
)

# Applications that already have a card at least once
apps_already_with_card = hist_df.loc[
    hist_df["card_number"].notna(),
    "application_number"
].drop_duplicates()

# Applications that had missing card earlier,
# but have NEVER received a card yet
apps_to_check = (
    hist_df.loc[
        hist_df["card_number"].isna()
        & ~hist_df["application_number"].isin(apps_already_with_card),
        "application_number"
    ]
    .drop_duplicates()
)

        new_records.append({
