start_date = dbutils.widgets.get("start_date")
end_date = dbutils.widgets.get("end_date")

print(f"Start Date: {start_date}")
print(f"End Date: {end_date}")

spark.sql(f"""
CREATE OR REPLACE TEMP VIEW pda_parameters AS

WITH base_dates AS (
    SELECT
        DATE('{start_date}') AS start_date,
        DATE('{end_date}') AS end_date
)

SELECT
    start_date,
    end_date,

    -- Previous-month MTD start
    TO_DATE(
        DATE_TRUNC(
            'MONTH',
            ADD_MONTHS(end_date, -1)
        )
    ) AS previous_month_start,

    -- Same day in previous month
    ADD_MONTHS(end_date, -1) AS previous_month_end,

    -- Current month start
    TO_DATE(
        DATE_TRUNC('MONTH', end_date)
    ) AS current_mtd_start,

    -- Last seven calendar days, including end_date
    DATE_SUB(end_date, 6) AS week_start,

    end_date AS week_end

FROM base_dates
""")

%sql

CREATE OR REPLACE TEMP VIEW pda_fico_base AS

SELECT
    REGEXP_REPLACE(
        CAST(fico.customerAcctNumber AS STRING),
        '-',
        ''
    ) AS agreement_number,

    CAST(fico.customerIdFromHeader AS STRING) AS party_number,

    TO_DATE(fico.openDate) AS fico_open_date

FROM cad_fico.ais fico
CROSS JOIN pda_parameters p

WHERE
    fico.openDate >= p.start_date
    AND fico.openDate < DATE_ADD(p.end_date, 1)
    AND CAST(fico.currencyCode AS STRING) = '124';


%sql

CREATE OR REPLACE TEMP VIEW pda_agreement_base AS

SELECT DISTINCT
    CAST(b.agmt_cd AS STRING) AS agreement_number,
    TO_DATE(b.agmt_open_dt) AS agreement_open_date

FROM edf.agmt_can b
CROSS JOIN pda_parameters p

WHERE
    b.appl_cd = 'PDA'
    AND b.agmt_open_dt >= p.start_date
    AND b.agmt_open_dt < DATE_ADD(p.end_date, 1);

%sql

CREATE OR REPLACE TEMP VIEW pda_applications AS

WITH application_candidates AS (
    SELECT
        -- Original final application-number construction
        CONCAT(
            'P',
            CAST(pa.agmt_cd AS STRING),
            '0',
            LEFT(CAST(pa.prty_cd AS STRING), 5)
        ) AS application_number,

        COALESCE(
            fico.fico_open_date,
            agreement.agreement_open_date
        ) AS application_date,

        TO_DATE(customer.cust_sinc_dt) AS customer_since_date,

        customer.eff_start_dt AS record_effective_date,

        ROW_NUMBER() OVER (
            PARTITION BY
                CONCAT(
                    'P',
                    CAST(pa.agmt_cd AS STRING),
                    '0',
                    LEFT(CAST(pa.prty_cd AS STRING), 5)
                )
            ORDER BY
                customer.eff_start_dt DESC NULLS LAST
        ) AS row_num

    FROM edf.prty_agmt_can pa

    INNER JOIN pda_agreement_base agreement
        ON CAST(pa.agmt_cd AS STRING)
         = agreement.agreement_number

    LEFT JOIN pda_fico_base fico
        ON CAST(pa.agmt_cd AS STRING)
         = fico.agreement_number
       AND CAST(pa.prty_cd AS STRING)
         = fico.party_number

    LEFT JOIN edf.prty_can customer
        ON CAST(pa.prty_cd AS STRING)
         = CAST(customer.prty_cd AS STRING)
       AND customer.is_active_ind = 'Y'

    WHERE
        COALESCE(
            fico.fico_open_date,
            agreement.agreement_open_date
        ) IS NOT NULL
),

deduplicated_applications AS (
    SELECT
        application_number,
        application_date,
        customer_since_date

    FROM application_candidates

    WHERE row_num = 1
)

SELECT
    application_number,
    application_date,
    customer_since_date,

    CASE
        WHEN customer_since_date IS NULL THEN 'UNKNOWN'

        WHEN DATEDIFF(
            application_date,
            customer_since_date
        ) <= 90
        THEN 'NTB'

        ELSE 'ETB'
    END AS customer_tenure_90d

FROM deduplicated_applications;



%sql

SELECT
    COUNT(*) AS row_count,
    COUNT(DISTINCT application_number) AS distinct_applications,
    MIN(application_date) AS minimum_application_date,
    MAX(application_date) AS maximum_application_date,
    SUM(CASE WHEN customer_tenure_90d = 'NTB' THEN 1 ELSE 0 END) AS ntb_apps,
    SUM(CASE WHEN customer_tenure_90d = 'ETB' THEN 1 ELSE 0 END) AS etb_apps,
    SUM(CASE WHEN customer_tenure_90d = 'UNKNOWN' THEN 1 ELSE 0 END) AS unknown_tenure_apps

FROM pda_applications;


%sql

CREATE OR REPLACE TEMP VIEW pda_fraud_actions AS

SELECT
    CAST(act.application_number AS STRING) AS application_number,
    MAX(act.date_updated) AS latest_fraud_action_date

FROM hive_metastore.fiqm.tdct_rt_action act
CROSS JOIN pda_parameters p

WHERE
    act.application_number LIKE 'P%D%'
    AND UPPER(TRIM(act.action_status_code)) IN ('H', 'S')

    AND act.date_updated >= p.start_date
    AND act.date_updated < DATE_ADD(p.end_date, 1)

GROUP BY
    CAST(act.application_number AS STRING);



%sql

CREATE OR REPLACE TEMP VIEW pda_application_fraud_base AS

SELECT
    app.application_number,
    app.application_date,
    app.customer_since_date,
    app.customer_tenure_90d,

    CASE
        WHEN fraud.application_number IS NOT NULL THEN 1
        ELSE 0
    END AS fraud_flag,

    fraud.latest_fraud_action_date

FROM pda_applications app

LEFT JOIN pda_fraud_actions fraud
    ON app.application_number = fraud.application_number;


%sql

WITH app_periods AS (

    -- ========================================================
    -- MONTHLY
    -- ========================================================
    SELECT
        'MONTHLY' AS reporting_period,
        DATE_FORMAT(application_date, 'yyyy-MM') AS period_label,
        COUNT(DISTINCT application_number) AS num_apps

    FROM pda_application_fraud_base

    GROUP BY
        DATE_FORMAT(application_date, 'yyyy-MM')


    UNION ALL


    -- ========================================================
    -- PREVIOUS MONTH MTD
    -- ========================================================
    SELECT
        'PREVIOUS_MONTH_MTD' AS reporting_period,

        CONCAT(
            DATE_FORMAT(p.previous_month_start, 'yyyy-MM-dd'),
            ' to ',
            DATE_FORMAT(p.previous_month_end, 'yyyy-MM-dd')
        ) AS period_label,

        COUNT(DISTINCT d.application_number) AS num_apps

    FROM pda_application_fraud_base d
    CROSS JOIN pda_parameters p

    WHERE
        d.application_date >= p.previous_month_start
        AND d.application_date < DATE_ADD(p.previous_month_end, 1)

    GROUP BY
        p.previous_month_start,
        p.previous_month_end


    UNION ALL


    -- ========================================================
    -- CURRENT MTD
    -- ========================================================
    SELECT
        'CURRENT_MTD' AS reporting_period,

        CONCAT(
            DATE_FORMAT(p.current_mtd_start, 'yyyy-MM-dd'),
            ' to ',
            DATE_FORMAT(p.end_date, 'yyyy-MM-dd')
        ) AS period_label,

        COUNT(DISTINCT d.application_number) AS num_apps

    FROM pda_application_fraud_base d
    CROSS JOIN pda_parameters p

    WHERE
        d.application_date >= p.current_mtd_start
        AND d.application_date < DATE_ADD(p.end_date, 1)

    GROUP BY
        p.current_mtd_start,
        p.end_date


    UNION ALL


    -- ========================================================
    -- WEEKLY
    -- ========================================================
    SELECT
        'WEEKLY' AS reporting_period,

        CONCAT(
            DATE_FORMAT(p.week_start, 'yyyy-MM-dd'),
            ' to ',
            DATE_FORMAT(p.week_end, 'yyyy-MM-dd')
        ) AS period_label,

        COUNT(DISTINCT d.application_number) AS num_apps

    FROM pda_application_fraud_base d
    CROSS JOIN pda_parameters p

    WHERE
        d.application_date >= p.week_start
        AND d.application_date < DATE_ADD(p.week_end, 1)

    GROUP BY
        p.week_start,
        p.week_end
),

fraud_periods AS (

    -- ========================================================
    -- MONTHLY
    -- ========================================================
    SELECT
        'MONTHLY' AS reporting_period,
        DATE_FORMAT(application_date, 'yyyy-MM') AS period_label,

        COUNT(
            DISTINCT CASE
                WHEN fraud_flag = 1 THEN application_number
            END
        ) AS num_frauds

    FROM pda_application_fraud_base

    GROUP BY
        DATE_FORMAT(application_date, 'yyyy-MM')


    UNION ALL


    -- ========================================================
    -- PREVIOUS MONTH MTD
    -- ========================================================
    SELECT
        'PREVIOUS_MONTH_MTD' AS reporting_period,

        CONCAT(
            DATE_FORMAT(p.previous_month_start, 'yyyy-MM-dd'),
            ' to ',
            DATE_FORMAT(p.previous_month_end, 'yyyy-MM-dd')
        ) AS period_label,

        COUNT(
            DISTINCT CASE
                WHEN d.fraud_flag = 1 THEN d.application_number
            END
        ) AS num_frauds

    FROM pda_application_fraud_base d
    CROSS JOIN pda_parameters p

    WHERE
        d.application_date >= p.previous_month_start
        AND d.application_date < DATE_ADD(p.previous_month_end, 1)

    GROUP BY
        p.previous_month_start,
        p.previous_month_end


    UNION ALL


    -- ========================================================
    -- CURRENT MTD
    -- ========================================================
    SELECT
        'CURRENT_MTD' AS reporting_period,

        CONCAT(
            DATE_FORMAT(p.current_mtd_start, 'yyyy-MM-dd'),
            ' to ',
            DATE_FORMAT(p.end_date, 'yyyy-MM-dd')
        ) AS period_label,

        COUNT(
            DISTINCT CASE
                WHEN d.fraud_flag = 1 THEN d.application_number
            END
        ) AS num_frauds

    FROM pda_application_fraud_base d
    CROSS JOIN pda_parameters p

    WHERE
        d.application_date >= p.current_mtd_start
        AND d.application_date < DATE_ADD(p.end_date, 1)

    GROUP BY
        p.current_mtd_start,
        p.end_date


    UNION ALL


    -- ========================================================
    -- WEEKLY
    -- ========================================================
    SELECT
        'WEEKLY' AS reporting_period,

        CONCAT(
            DATE_FORMAT(p.week_start, 'yyyy-MM-dd'),
            ' to ',
            DATE_FORMAT(p.week_end, 'yyyy-MM-dd')
        ) AS period_label,

        COUNT(
            DISTINCT CASE
                WHEN d.fraud_flag = 1 THEN d.application_number
            END
        ) AS num_frauds

    FROM pda_application_fraud_base d
    CROSS JOIN pda_parameters p

    WHERE
        d.application_date >= p.week_start
        AND d.application_date < DATE_ADD(p.week_end, 1)

    GROUP BY
        p.week_start,
        p.week_end
)

SELECT
    a.reporting_period,
    a.period_label,
    a.num_apps,
    COALESCE(f.num_frauds, 0) AS num_frauds,

    ROUND(
        100.0 * COALESCE(f.num_frauds, 0)
        / NULLIF(a.num_apps, 0),
        2
    ) AS fraud_rate_pct

FROM app_periods a

LEFT JOIN fraud_periods f
    ON a.reporting_period = f.reporting_period
   AND a.period_label = f.period_label

ORDER BY
    CASE a.reporting_period
        WHEN 'MONTHLY' THEN 1
        WHEN 'PREVIOUS_MONTH_MTD' THEN 2
        WHEN 'CURRENT_MTD' THEN 3
        WHEN 'WEEKLY' THEN 4
        ELSE 5
    END,
    a.period_label;
