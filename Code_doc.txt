tdct_perf_sql = """
WITH alerts AS (
    SELECT *
    FROM hive_metastore.fiqm.tdct_rt_match
    WHERE RULE_DISPLAYNAME = 'Visa App Manual Alerts'
      AND CREATE_DATE_TIME >= '2026-06-03'
),
actions AS (
    SELECT 
        a.*,
        b.ACTION_STATUS_CODE,
        ROW_NUMBER() OVER (
            PARTITION BY a.APPLICATION_NUMBER
            ORDER BY b.DATE_UPDATED DESC
        ) AS rn
    FROM alerts a
    LEFT JOIN hive_metastore.fiqm.tdct_rt_action b
        ON a.APPLICATION_NUMBER = b.APPLICATION_NUMBER
)
SELECT *
FROM actions
WHERE rn = 1
"""

mbna_perf_sql = """
WITH alerts AS (
    SELECT *
    FROM hive_metastore.fiqm.tdct_rt_match
    WHERE TASK_NAME LIKE '%MBNA_QUETS - T1 - Manual Alerts%'
      AND CREATE_DATE_TIME >= '2026-06-03'
),
actions AS (
    SELECT 
        a.*,
        b.ACTION_STATUS_CODE,
        ROW_NUMBER() OVER (
            PARTITION BY a.APPLICATION_NUMBER
            ORDER BY b.DATE_UPDATED DESC
        ) AS rn
    FROM alerts a
    LEFT JOIN hive_metastore.fiqm.tdct_rt_action b
        ON a.APPLICATION_NUMBER = b.APPLICATION_NUMBER
)
SELECT *
FROM actions
WHERE rn = 1
"""

tdct_base = spark.sql(tdct_perf_sql)
mbna_base = spark.sql(mbna_perf_sql)

tdct_base.createOrReplaceTempView("tdct_base")
mbna_base.createOrReplaceTempView("mbna_base")


weekly_perf = spark.sql("""
SELECT 
    portfolio,
    week_start_dt,
    num_alerts,
    num_frauds,
    num_suspects,
    ROUND((num_alerts - num_frauds) * 1.0 / NULLIF(num_frauds, 0), 2) AS fpr
FROM (
    SELECT
        'TDCT' AS portfolio,
        DATE_TRUNC('week', CREATE_DATE_TIME) AS week_start_dt,
        COUNT(DISTINCT APPLICATION_NUMBER) AS num_alerts,
        COUNT(DISTINCT CASE WHEN ACTION_STATUS_CODE IN ('H','S') THEN APPLICATION_NUMBER END) AS num_frauds,
        COUNT(DISTINCT CASE WHEN ACTION_STATUS_CODE = 'I' THEN APPLICATION_NUMBER END) AS num_suspects
    FROM tdct_base
    GROUP BY DATE_TRUNC('week', CREATE_DATE_TIME)

    UNION ALL

    SELECT
        'MBNA' AS portfolio,
        DATE_TRUNC('week', CREATE_DATE_TIME) AS week_start_dt,
        COUNT(DISTINCT APPLICATION_NUMBER) AS num_alerts,
        COUNT(DISTINCT CASE WHEN ACTION_STATUS_CODE IN ('H','S') THEN APPLICATION_NUMBER END) AS num_frauds,
        COUNT(DISTINCT CASE WHEN ACTION_STATUS_CODE = 'I' THEN APPLICATION_NUMBER END) AS num_suspects
    FROM mbna_base
    GROUP BY DATE_TRUNC('week', CREATE_DATE_TIME)
)
ORDER BY portfolio, week_start_dt
""")


daily_perf = spark.sql("""
SELECT 
    portfolio,
    alert_dt,
    num_alerts,
    num_frauds,
    num_suspects,
    ROUND((num_alerts - num_frauds) * 1.0 / NULLIF(num_frauds, 0), 2) AS fpr
FROM (
    SELECT
        'TDCT' AS portfolio,
        DATE(CREATE_DATE_TIME) AS alert_dt,
        COUNT(DISTINCT APPLICATION_NUMBER) AS num_alerts,
        COUNT(DISTINCT CASE WHEN ACTION_STATUS_CODE IN ('H','S') THEN APPLICATION_NUMBER END) AS num_frauds,
        COUNT(DISTINCT CASE WHEN ACTION_STATUS_CODE = 'I' THEN APPLICATION_NUMBER END) AS num_suspects
    FROM tdct_base
    WHERE DATE(CREATE_DATE_TIME) >= DATE_SUB(CURRENT_DATE(), 5)
    GROUP BY DATE(CREATE_DATE_TIME)

    UNION ALL

    SELECT
        'MBNA' AS portfolio,
        DATE(CREATE_DATE_TIME) AS alert_dt,
        COUNT(DISTINCT APPLICATION_NUMBER) AS num_alerts,
        COUNT(DISTINCT CASE WHEN ACTION_STATUS_CODE IN ('H','S') THEN APPLICATION_NUMBER END) AS num_frauds,
        COUNT(DISTINCT CASE WHEN ACTION_STATUS_CODE = 'I' THEN APPLICATION_NUMBER END) AS num_suspects
    FROM mbna_base
    WHERE DATE(CREATE_DATE_TIME) >= DATE_SUB(CURRENT_DATE(), 5)
    GROUP BY DATE(CREATE_DATE_TIME)
)
ORDER BY portfolio, alert_dt
""")


weekly_perf_html = weekly_perf.toPandas().to_html(index=False)
