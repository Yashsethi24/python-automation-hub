import pandas as pd

X = custom_df.drop(columns=['Fraud_ind', 'Alert_ind'])
y = custom_df['Fraud_ind']

binary_cols = [
    'IsCustomerEmailPopularityLow',
    'IsCustomerEmailPopularityMedium',
    'IsCustomerEmailPopularityHigh',
    'IsCustomerEmailVelocityLow',
    'IsCustomerEmailVelocityMedium',
    'IsCustomerEmailVelocityHigh',
    'IsCustomerEmailAgeLow',
    'IsCustomerEmailAgeMedium',
    'IsCustomerEmailAgeHigh'
]

for col in binary_cols:
    X[col] = X[col].astype(int)

X['IPCountry'] = X['IPCountry'].fillna('UNKNOWN')
X['IPCarrier'] = X['IPCarrier'].fillna('UNKNOWN')
X['IPRegion'] = X['IPRegion'].fillna('UNKNOWN')

top = X['IPCarrier'].value_counts().nlargest(20).index

X['IPCarrier'] = X['IPCarrier'].where(
    X['IPCarrier'].isin(top),
    'OTHER'
)

X['EmailDomain'] = (
    X['CustomerEmails']
      .str.split('@')
      .str[-1]
      .str.lower()
)

X['PostalPrefix'] = X['CustomerStdAddressPostalCode'].str[:3]

X['AreaCode'] = (
    X['CustomerPhoneNumber']
      .astype(str)
      .str[:3]
)

device_freq = X['DeviceTag'].value_counts()

X['DeviceFrequency'] = X['DeviceTag'].map(device_freq)



cat_cols = X.select_dtypes(include='object').columns

X = pd.get_dummies(
    X,
    columns=cat_cols,
    dummy_na=True
)

from sklearn.tree import DecisionTreeClassifier

tree = DecisionTreeClassifier(
    max_depth=4,
    min_samples_leaf=30,
    class_weight='balanced',
    random_state=42
)


tree.fit(X, y)

from sklearn.tree import export_text

rules = export_text(
    tree,
    feature_names=list(X.columns)
)

print(rules)

leaf = tree.apply(X)

summary = (
    pd.DataFrame({
        'Leaf': leaf,
        'Fraud': y
    })
    .groupby('Leaf')
    .agg(
        Total=('Fraud','size'),
        Fraud=('Fraud','sum')
    )
)

summary['Genuine'] = summary['Total'] - summary['Fraud']
summary['Precision'] = summary['Fraud'] / summary['Total']

summary.sort_values(
    'Precision',
    ascending=False
)
