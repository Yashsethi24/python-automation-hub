import os
import re
import glob

folder_path = r"C:\Your\Folder"

# Step 1: Delete existing CON_EXP files
for file in glob.glob(os.path.join(folder_path, "CON_EXP*.xlsx")):
    os.remove(file)
    print(f"Deleted: {os.path.basename(file)}")

# Step 2: Rename new files
for filename in os.listdir(folder_path):

    if not filename.endswith(".xlsx"):
        continue

    match = re.search(r'EXP(\d+)', filename, re.IGNORECASE)

    if match:
        exp_num = match.group(1)

        new_filename = f"CON_EXP{exp_num}.xlsx"

        old_path = os.path.join(folder_path, filename)
        new_path = os.path.join(folder_path, new_filename)

        os.rename(old_path, new_path)

        print(f"Renamed: {filename} -> {new_filename}")
