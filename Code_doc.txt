import pandas as pd
import os
import glob

folder_path = r"C:\Your\Folder"

df_list = []

for file in glob.glob(os.path.join(folder_path, "*.txt")):
    df = pd.read_csv(
        file,
        sep="|",
        dtype=str,          # Read everything as string
        low_memory=False
    )

    # Optional: Add source filename
    df["source_file"] = os.path.basename(file)

    df_list.append(df)

# Concatenate all files
final_df = pd.concat(df_list, ignore_index=True)

print(f"Files read: {len(df_list)}")
print(f"Total rows: {len(final_df)}")
