import pandas as pd
import glob
import os

folder_path = r"C:\Your\Folder"

dfs = {}

for file in glob.glob(os.path.join(folder_path, "*.csv")):
    filename = os.path.basename(file)

    # Extract text after 'pda_' and before '.csv'
    key = filename.split("pda_")[1].replace(".csv", "")

    dfs[key] = pd.read_csv(file)

    print(f"Loaded {filename} as dfs['{key}']")
