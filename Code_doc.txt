%sql

WITH alerts AS
(
    SELECT *
    FROM hive_metastore.fiqm.tdct_rt_batch
    WHERE RULE_DISPLAYNAME = 'Visa App Manual Alerts'
      AND CREATE_DATE_TIME >= '2026-06-03'
),

actions AS
(
    SELECT 
        a.*,
        b.ACTION_STATUS_CODE,
        ROW_NUMBER() OVER (
            PARTITION BY a.APPLICATION_NUMBER
            ORDER BY b.DATE_UPDATED DESC
        ) AS rn
    FROM alerts a
    LEFT JOIN hive_metastore.fiqm.tdct_rt_action b
        ON a.APPLICATION_NUMBER = b.APPLICATION_NUMBER
)

SELECT
    DATE_TRUNC('week', CREATE_DATE_TIME) AS week_start_dt,
    COUNT(DISTINCT APPLICATION_NUMBER) AS num_alerts,

    COUNT(DISTINCT CASE
        WHEN ACTION_STATUS_CODE IN ('H','S')
        THEN APPLICATION_NUMBER
    END) AS num_frauds,

    COUNT(DISTINCT CASE
        WHEN ACTION_STATUS_CODE = 'I'
        THEN APPLICATION_NUMBER
    END) AS num_suspects,

    ROUND(
        (
            COUNT(DISTINCT APPLICATION_NUMBER)
            - COUNT(DISTINCT CASE
                WHEN ACTION_STATUS_CODE IN ('H','S')
                THEN APPLICATION_NUMBER
            END)
        ) * 1.0
        /
        NULLIF(
            COUNT(DISTINCT CASE
                WHEN ACTION_STATUS_CODE IN ('H','S')
                THEN APPLICATION_NUMBER
            END),
            0
        ),
        2
    ) AS FPR

FROM actions
WHERE rn = 1
GROUP BY DATE_TRUNC('week', CREATE_DATE_TIME)
ORDER BY week_start_dt;
